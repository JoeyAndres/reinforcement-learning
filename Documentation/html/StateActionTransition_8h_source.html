126 template<class S>
127 void StateActionTransition<S>::update(const S& nextState,
128  const AI::FLOAT reward) {
129  boost::unique_lock < boost::shared_mutex > rewardLock(_rewardMutex);
130  boost::unique_lock < boost::shared_mutex > frequencyLock(_frequencyMutex);
131 
132  _stateActionTransitionFrequency.insert(
133  std::pair<S, AI::FLOAT>(nextState, 0.0F));
134  _stateActionTransitionReward.insert(
135  std::pair<S, AI::FLOAT>(nextState, reward));
136 
137  // Update Frequency.
138  // --Lower the value of all other frequencies.
139  for (auto iter = _stateActionTransitionFrequency.begin();
140  iter != _stateActionTransitionFrequency.end(); iter++) {
141  const S& state = iter->first;
142  if (state != nextState) {
143  _stateActionTransitionFrequency[state] =
144  _stateActionTransitionFrequency[state]
145  + _stepSize * (0.0F - _stateActionTransitionFrequency[state]);
146  }
147  }
148 
149  // --Increase the value of the next frequency.
150  _stateActionTransitionFrequency[nextState] =
151  _stateActionTransitionFrequency[nextState]
152  + _stepSize * (1.0F - _stateActionTransitionFrequency[nextState]);
153 
154  // Note, no need to lower the reward. Updating the reward will suffice.
155  _stateActionTransitionReward[nextState] =
156  _stateActionTransitionReward[nextState]
157  + 1.0F * (reward - _stateActionTransitionReward[nextState]);
158 
159  assert(
160  _stateActionTransitionFrequency.size()
161  == _stateActionTransitionReward.size());
162 }

185 template<class S>
186 const S& StateActionTransition<S>::getNextState() const
187  throw (StateActionTransitionException) {
188  boost::shared_lock < boost::shared_mutex > frequencyLock(_frequencyMutex);
189  StateActionTransitionException exception(
190  "StateActionTransition<S, AI::FLOAT>::getNextState(): nextStates are empty.");
191  if (_stateActionTransitionFrequency.size() == 0) {
192  throw exception;
193  }
194 
195  // Get a random number in [0.0,freqValSum].
196  std::uniform_real_distribution<AI::FLOAT> distributionRandomSelection(0.0F,
197  1.0F);
198  AI::FLOAT randomNumberRandomSelection = distributionRandomSelection(
199  _randomDevice);
200 
201  // If randomNumberRandomSelection > greediness, return a random state.
202  if (randomNumberRandomSelection > _greedy) {
203  auto it = _stateActionTransitionFrequency.begin();
204  std::advance(it, _randomDevice() % _stateActionTransitionFrequency.size());
205  const S& nextState = it->first;
206  return nextState;
207  }
208 
209  // http://stackoverflow.com/questions/1761626/weighted-random-numbers
210  // Acquire the sum of all the frequency values.
211  AI::FLOAT freqValSum = 0.0F;
212  for (auto iter = _stateActionTransitionFrequency.begin();
213  iter != _stateActionTransitionFrequency.end(); iter++) {
214  freqValSum += iter->second;
215  }
216 
217  // Get a random number in [0.0,freqValSum] for weighted selection.
218  std::uniform_real_distribution<AI::FLOAT> distributionWeightedSelection(
219  0.0F, freqValSum);
220  AI::FLOAT randomNumberWeightedSelection = distributionWeightedSelection(
221  _randomDevice);
222 
223  // Get the one that matches the random value.
224  auto iter = _stateActionTransitionFrequency.begin();
225  for (; iter != _stateActionTransitionFrequency.end(); iter++) {
226  AI::FLOAT currentFreq = iter->second;
227 
228  // If current frequency is greater than highest frequency and at the
229  // time lower than the current random number then
230  // Error = randomNumber-highestFreq is of the lowest, therefore,
231  // a candidate.
232  if (currentFreq > randomNumberWeightedSelection) {
233  return iter->first;
234  }
235 
236  randomNumberWeightedSelection -= currentFreq;
237  }
238 
239  assert(false); // Note supposed to reach here.
240 }
