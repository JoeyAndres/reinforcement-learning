79 const A& AI::Algorithm::Policy::Softmax<S, A>::getAction(
80  const map<A, AI::FLOAT>& actionValues, const set<A>& actionSet) {
81  // Acquire E(i=1...n) e^(Q(i)/temp)
82  AI::FLOAT sum = 0.0F;
83  for (const A& action : actionSet) {
84  sum += exp(actionValues.at(action) / _temperature);
85  }
86 
87  // Acquire probability for each action.
88  map<A, AI::FLOAT> actionProbabilityMap;
89  for (const A& action : actionSet) {
90  AI::FLOAT probability = exp(actionValues.at(action) / _temperature) / sum;
91  actionProbabilityMap[action] = probability;
92  }
93 
94  // Choose action based on probability.
95  AI::FLOAT randomNumberWeightedSelection = _distribution(_randomDevice);
96 
97  // Get the one that matches the random value.
98  auto iter = actionProbabilityMap.begin();
99  for (; iter != actionProbabilityMap.end(); iter++) {
100  AI::FLOAT currentProbability = iter->second;
101 
102  // If current frequency is greater than highest frequency and at the
103  // time lower than the current random number then
104  // Error = randomNumber-highestFreq is of the lowest, therefore,
105  // a candidate.
106  if (currentProbability > randomNumberWeightedSelection) {
107  return iter->first;
108  }
109 
110  randomNumberWeightedSelection -= currentProbability;
111  }
112 
113  assert(false);
114 }
