142 template<class S, class A>
143 void AI::Agent<S, A>::execute() {
144  _actuatorInstance.applyAction(_currentAction);
145  S nextState = _getCurrentState();
146  FLOAT reward = _sensorInstance.getReward(_currentState);
147 
148  // Accumulate reward.
149  _accumulativeReward += reward;
150 
151  // Update value, e.g. perform back ups.
152  _learningAlgorithm.update(StateAction<S, A>(_currentState, _currentAction),
153  nextState, reward,
154  _actuatorInstance.getActionSet());
155 
156  // Get the action from the algorithm.
157  // Note the algorithm will retrieve from controlPolicy.
158  _currentAction = _learningAlgorithm.getAction(
159  nextState, _actuatorInstance.getActionSet());
160 
161  // Update current state.
162  _currentState = nextState;
163 }
