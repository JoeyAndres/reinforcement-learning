126 void AI::Agent<S, A>::execute() {
127  _actuatorInstance.applyAction(_currentAction);
128  S nextState = _getCurrentState();
129  FLOAT reward = _sensorInstance.getReward(_currentState);
130 
131  // Accumulate reward.
132  _accumulativeReward += reward;
133 
134  // Update value, e.g. perform back ups.
135  _learningAlgorithm.update(StateAction<S, A>(_currentState, _currentAction),
136  nextState, reward,
137  _actuatorInstance.getActionSet());
138 
139  // Get the action from the algorithm.
140  // *Note the algorithm will retrieve from controlPolicy.
141  _currentAction = _learningAlgorithm.getAction(
142  nextState, _actuatorInstance.getActionSet());
143 
144  // Update current state.
145  _currentState = nextState;
146 }
