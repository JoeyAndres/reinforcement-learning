102 template<class S, class A>
103 inline void AI::Algorithm::DynaQCCBase<S, A>::_threadWorker(
104  const set<A>& actionSet, AI::UINT localSimulationSteps) {
105  if (this->_model.empty())
106  return;
107  for (AI::UINT i = 0; i < localSimulationSteps; i++) {
108  // Pick a random state in _model.
109  this->_modelLock.lock();
110  auto item = this->_model.begin();
111  std::advance(item, _distribution(_randomDevice));
112  const StateActionTransition<S>& st = this->_model.at(item->first);
113  const S& transState = st.getNextState();
114  this->_modelLock.unlock();
115 
116  A nextAction = this->argMax(transState, actionSet);
117 
118  this->backUpStateActionPair(item->first, st.getReward(transState),
119  StateAction<S, A>(transState, nextAction));
120  }
121 }
