59 template<class S, class A>
60 inline void AI::Algorithm::DynaQCCBase<S, A>::_simulate(
61  const set<A>& actionSet) {
62  std::vector < boost::thread > threadVector;
63  AI::UINT localSimulationSteps = this->_simulationIterationCount / 12;
64  _updateDistribution();
65  for (AI::INT i = 0; i < 12; i++) {
66  threadVector.push_back(
67  boost::thread(&DynaQCCBase<S, A>::_threadWorker, this, actionSet,
68  localSimulationSteps));
69  }
70 
71  for (boost::thread& t : threadVector) {
72  t.join();
73  }
74  threadVector.clear();
75 }

77 template<class S, class A>
78 inline void AI::Algorithm::DynaQCCBase<S, A>::_threadWorker(
79  const set<A>& actionSet, AI::UINT localSimulationSteps) {
80  if (this->_model.empty())
81  return;
82  for (AI::UINT i = 0; i < localSimulationSteps; i++) {
83  // Pick a random state in _model.
84  this->_modelLock.lock();
85  auto item = this->_model.begin();
86  std::advance(item, _distribution(_randomDevice));
87  const StateActionTransition<S>& st = this->_model.at(item->first);
88  const S& transState = st.getNextState();
89  this->_modelLock.unlock();
90 
91  A nextAction = this->argMax(transState, actionSet);
92 
93  this->backUpStateActionPair(item->first, st.getReward(transState),
94  StateAction<S, A>(transState, nextAction));
95  }
96 }
