153 template<class S, class A>
154 A AI::Algorithm::ReinforcementLearning<S, A>::argMax(
155  const S& state, const set<A>& actionSet) const {
156  A greedAct = *(actionSet.begin());
157  AI::FLOAT currentValue = this->_stateActionPairContainer[StateAction<S, A>(
158  state, greedAct)];
159  for (const A& action : actionSet) {
160  AI::FLOAT value = this->_stateActionPairContainer[StateAction<S, A>(state,
161  action)];
162  if (currentValue < value) {
163  currentValue = value;
164  greedAct = action;
165  }
166  }
167  return greedAct;
168 }

246 template<class S, class A>
247 void AI::Algorithm::ReinforcementLearning<S, A>::backUpStateActionPair(
248  const StateAction<S, A>& currentStateAction, const AI::FLOAT reward,
249  const StateAction<S, A>& nextStateActionPair) {
250  boost::unique_lock<boost::shared_mutex> containerLock(_containerLock);
251 
252  // Next state-action value (nSAV) and current state-action value (cSAV).
253  AI::FLOAT nSAV = getStateActionValue(nextStateActionPair);
254  AI::FLOAT cSAV = getStateActionValue(currentStateAction);
255  AI::FLOAT discountRate = this->_discountRate;
256 
257  setStateActionValue(
258  currentStateAction,
259  cSAV + this->_stepSize * (reward + discountRate * nSAV - cSAV));
260 }
