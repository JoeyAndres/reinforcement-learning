130 template<class S, class A>
131 A AI::Algorithm::ReinforcementLearning<S, A>::argMax(
132  const S& state, const set<A>& actionSet) const {
133  A greedAct = *(actionSet.begin());
134  AI::FLOAT currentValue = this->_stateActionPairContainer[StateAction<S, A>(
135  state, greedAct)];
136  for (const A& action : actionSet) {
137  AI::FLOAT value = this->_stateActionPairContainer[StateAction<S, A>(state,
138  action)];
139  if (currentValue < value) {
140  currentValue = value;
141  greedAct = action;
142  }
143  }
144  return greedAct;
145 }

224 void AI::Algorithm::ReinforcementLearning<S, A>::backUpStateActionPair(
225  const StateAction<S, A>& currentStateAction, const AI::FLOAT reward,
226  const StateAction<S, A>& nextStateActionPair) {
227  boost::unique_lock < boost::shared_mutex > containerLock(_containerLock);
228 
229  setStateActionValue(
230  currentStateAction,
231  getStateActionValue(currentStateAction)
232  + this->_stepSize

234  + this->_discountRate
235  * getStateActionValue(nextStateActionPair)
236  - getStateActionValue(currentStateAction)));
