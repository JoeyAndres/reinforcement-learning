91 const A& AI::Algorithm::Policy::EpsilonGreedy<S, A>::getAction(
92  const map<A, AI::FLOAT>& actionValues, const set<A>& actionSet) {
93  const AI::FLOAT& r = _distribution(_randomDevice);
94  if (r > _greediness) {
95  uniform_int_distribution<AI::INT> indexDistribution(0, actionSet.size());
96  typename set<A>::const_iterator it(actionSet.begin());
97  advance(it, indexDistribution(_randomDevice));
98  return (*it);
99  } else {
100  return argMax(actionValues, actionSet);
101  }
102 }
