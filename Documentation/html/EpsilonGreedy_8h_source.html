73 const A& AI::Algorithm::EpsilonGreedy<S, A>::getAction(
74  const map<A, AI::FLOAT>& actionValues, const set<A>& actionSet) {
75  const AI::FLOAT& r = _distribution(_randomDevice);
76  if (r > _greediness) {
77  uniform_int_distribution<AI::INT> indexDistribution(0, actionSet.size());
78  typename set<A>::const_iterator it(actionSet.begin());
79  advance(it, indexDistribution(_randomDevice));
80  return (*it);
81  } else {
82  return argMax(actionValues, actionSet);
83  }
84 }
