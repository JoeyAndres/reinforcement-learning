93 template<class S, class A>
94 void DynaQPrioritizeSweeping<S, A>::_prioritySweep(const set<A>& actionSet) {
95  // Repeat n times while priority queue is not empty.
96  for (AI::UINT i = 0; i < this->_simulationIterationCount; i++) {
97  if (_priorityQueue.empty())
98  break;
99 
100  StateAction<S, A> currentStateActionPair = _priorityQueue.top().first;
101  _priorityQueue.pop();
102 
103  // Acquire next reward from model.
104  const StateActionTransition<S>& currentStateActionTransition = this->_model
105  .at(currentStateActionPair);
106  const S& nextState = currentStateActionTransition.getNextState();
107  AI::FLOAT nextReward = currentStateActionTransition.getReward(nextState);
108 
109  // Acquire max_action(Q(S, a)).
110  A nextAction = this->getLearningAction(nextState, actionSet);
111 
112  StateAction<S, A> nextStateAction(nextState, nextAction);
113 
114  // Update Q(S, A).
115  this->backUpStateActionPair(currentStateActionPair, nextReward,
116  nextStateAction);
117 
118  for (typename map<StateAction<S, A>, StateActionTransition<S> >::iterator iter =
119  this->_model.begin(); iter != this->_model.end(); iter++) {
120  // Acquire a model(S) and look for a model(S) that transition to current(S, A).
121  StateActionTransition<S>& modelStateTransition = iter->second;
122  const S& modelNextState = modelStateTransition.getNextState();
123 
124  // If model(S) -> current(S, A), that means model(S) factors in reaching terminal state.
125  // And because of that, back up model(S).
126  if (modelNextState == currentStateActionPair.getState()) {
127  AI::FLOAT priorReward = modelStateTransition.getReward(modelNextState);
128  A nextModelAction = this->argMax(modelNextState, actionSet);
129 
130  AI::FLOAT temptdError = _getTDError(
131  iter->first, priorReward,
132  StateAction<S, A>(modelNextState, nextModelAction));
133 
134  AI::FLOAT tempPriority = abs(temptdError);
135  if (tempPriority > _priorityThreshold) {
136  _priorityQueue.push(StateActionPair(iter->first, tempPriority));
137  }
138  }
139  }
140  }
141 }

144 AI::FLOAT DynaQPrioritizeSweeping<S, A>::_getTDError(
145  const StateAction<S, A>& currentStateAction, const AI::FLOAT reward,
146  const StateAction<S, A>& nextStateAction) {
147  AI::FLOAT tdError = (reward
148  + this->_discountRate * this->_stateActionPairContainer[nextStateAction]
149  - this->_stateActionPairContainer[currentStateAction]);
