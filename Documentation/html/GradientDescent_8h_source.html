212 void GradientDescent::updateWeights(
213  const vector<FLOAT>& stateVector, const actionVector<FLOAT>& av,
214  const vector<FLOAT>& nextState, const FLOAT reward,
215  const set<actionVector<FLOAT> >& actionSet) {
216  map<actionVector<FLOAT>, FLOAT> actionValueMap;
217  buildActionValues(actionSet, nextState, actionValueMap);
218  FLOAT maxValue = getMaxValue(actionValueMap);
219 
220  vector<FLOAT> stateVectorCopy = stateVector;
221  for (const FLOAT& action : av) {
222  stateVectorCopy.push_back(action);
223  }
224 
225  FEATURE_VECTOR fv;
226  getFeatureVector(stateVectorCopy, fv);
227  incrementEligibilityTraces(fv);
228 
229  FLOAT tdError = reward + _discountRate * maxValue
230  - getValueFromFeatureVector(fv);
231 
232  AI::FLOAT multiplier = _stepSize * tdError;
233  for (size_t i = 0; i < getSize(); i++) {
234  _w[i] += multiplier * _e[i];
235  }
236 
237  // Decrease traces.
238  decreaseEligibilityTraces();
239 }
