137 void GradientDescent::updateWeights(
138  const vector<FLOAT>& stateVector, const actionVector<FLOAT>& av,
139  const vector<FLOAT>& nextState, const FLOAT reward,
140  const set<actionVector<FLOAT> >& actionSet) {
141  map<actionVector<FLOAT>, FLOAT> actionValueMap;
142  buildActionValues(actionSet, nextState, actionValueMap);
143  FLOAT maxValue = getMaxValue(actionValueMap);
144 
145  vector<FLOAT> stateVectorCopy = stateVector;
146  for (const FLOAT& action : av) {
147  stateVectorCopy.push_back(action);
148  }
149 
150  FEATURE_VECTOR fv;
151  getFeatureVector(stateVectorCopy, fv);
152  incrementEligibilityTraces(fv);
153 
154  FLOAT tdError = reward + _discountRate * maxValue
155  - getValueFromFeatureVector(fv);
156 
157  AI::FLOAT multiplier = _stepSize * tdError;
158  for (size_t i = 0; i < getSize(); i++) {
159  _w[i] += multiplier * _e[i];
160  }
161 
162  // Decrease traces.
163  decreaseEligibilityTraces();
164 }
