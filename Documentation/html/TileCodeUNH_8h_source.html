80 inline void AI::Algorithm::TileCodeUNH::getFeatureVector(
81  const STATE_CONT& parameters, FEATURE_VECTOR& tilings) {
82  assert(this->_dimension == parameters.size());
83 
84  vector<AI::UINT> base(this->_dimension, 0);
85  vector<AI::INT> qStates(this->_dimension);
86  for (size_t i = 0; i < this->_dimension; i++) {
87  // Note to floor since casting to integer is not consistent
88  // with negative number. Casting is always a number toward zero.
89  qStates[i] = floor(parameters[i] * _normalization[i]);
90  }
91 
92  for (size_t i = 0; i < this->_numTilings; i++) {
93  vector<AI::INT> tileComponents(this->_dimension + 1);
94 
95  for (size_t j = 0; j < this->_dimension; j++) {
96  tileComponents[j] = qStates[j]
97  - mod(qStates[j] - base[j], this->_numTilings);
98  base[j] += 1 + (j * 2);
99  }
100  tileComponents[this->_dimension] = i;
101  AI::Algorithm::Hash::UNH hashAlg;
102  tilings.push_back(hashAlg.hash((AI::BYTE*)&tileComponents[0], this->_sizeCache));
103  }
104 }
