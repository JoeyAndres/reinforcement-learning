76 inline void AI::Algorithm::TileCodeUNH::getFeatureVector(
77  const STATE_CONT& parameters, FEATURE_VECTOR& tilings) {
78  assert(this->_dimension == parameters.size());
79 
80  vector<AI::UINT> base(this->_dimension, 0);
81  vector<AI::INT> qStates(this->_dimension);
82  for (size_t i = 0; i < this->_dimension; i++) {
83  // Note to floor since casting to integer is not consistent
84  // with negative number. Casting is always a number toward zero.
85  qStates[i] = floor(parameters[i] * _normalization[i]);
86  }
87 
88  for (size_t i = 0; i < this->_numTilings; i++) {
89  vector<AI::INT> tileComponents(this->_dimension + 1);
90 
91  for (size_t j = 0; j < this->_dimension; j++) {
92  tileComponents[j] = qStates[j]
93  - mod(qStates[j] - base[j], this->_numTilings);
94  base[j] += 1 + (j * 2);
95  }
96  tileComponents[this->_dimension] = i;
97  AI::Algorithm::Hash::UNH hashAlg;
98  tilings.push_back(hashAlg.hash((AI::BYTE*)&tileComponents[0], this->_sizeCache));
99  }
100 }
