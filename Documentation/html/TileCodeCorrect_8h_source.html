50  const STATE_CONT& parameters, FEATURE_VECTOR& tilings) {
51  assert(this->_dimension == parameters.size());
52 
53  vector<AI::INT> tileComponents(this->_dimension + 1);
54  for (AI::INT i = 0; i < this->_numTilings; i++) {
55  for (size_t j = 0; j < this->_dimension; j++) {
56  tileComponents[j] = this->_paramToGridValue(parameters[j], i, j);
57  }
58 
59  // x1 + x2*x1.gridSize + x3*x1.gridSize*x2.gridSize + ...
60  size_t hashedIndex = 0;
61  size_t tileComponentMultiplier = 1;
62  for (size_t j = 0; j < this->_dimension; j++) {
63  hashedIndex += tileComponents[j] * tileComponentMultiplier;
64  tileComponentMultiplier *= this->_dimensionalInfos[j].GetGridCountReal();
65  }
66 
67  hashedIndex += tileComponentMultiplier * i;
