96 TileCode::TileCode(vector<DimensionInfo<FLOAT> > dimensionalInfos,
97  size_t numTilings) {
98  assert(numTilings > 0);
99 
100  _dimensionalInfos = dimensionalInfos;
101  _dimension = dimensionalInfos.size();
102  _numTilings = numTilings;
103 
104  // Calculate the size.
105  _sizeCache = _calculateSizeCache();
106 
107  // Calculate random offsets.
108  std::uniform_real_distribution<AI::FLOAT> distribution(0, 1.0F);
109  for (size_t i = 0; i < _numTilings; i++) {
110  _randomOffsets.push_back(vector<AI::FLOAT>());
111  for (size_t j = 0; j < _dimension; j++) {
112  _randomOffsets[i].push_back(
113  distribution(_pseudoRNG) * _dimensionalInfos[j].GetOffsets()
114  * _dimensionalInfos[j].getGeneralizationScale());
115  }
116  }
117 }
