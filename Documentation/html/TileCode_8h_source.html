115 TileCode::TileCode(vector<DimensionInfo<FLOAT> > dimensionalInfos,
116  size_t numTilings) {
117  assert(numTilings > 0);
118 
119  _dimensionalInfos = dimensionalInfos;
120  _dimension = dimensionalInfos.size();
121  _numTilings = numTilings;
122 
123  // Calculate the size.
124  _sizeCache = _calculateSizeCache();
125 
126  // Calculate random offsets.
127  std::uniform_real_distribution<AI::FLOAT> distribution(0, 1.0F);
128  for (size_t i = 0; i < _numTilings; i++) {
129  _randomOffsets.push_back(vector<AI::FLOAT>());
130  for (size_t j = 0; j < _dimension; j++) {
131  _randomOffsets[i].push_back(
132  distribution(_pseudoRNG) * _dimensionalInfos[j].GetOffsets()
133  * _dimensionalInfos[j].getGeneralizationScale());
134  }
135  }
136 }
