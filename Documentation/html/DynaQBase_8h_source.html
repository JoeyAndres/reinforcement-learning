160 template<class S, class A>
161 void AI::Algorithm::DynaQBase<S, A>::_simulate(const set<A>& actionSet) {
162  if (_model.empty())
163  return;
164 
165  for (AI::UINT i = 0; i < _simulationIterationCount; i++) {
166  // Pick a random state in _model.
167  typename map<StateAction<S, A>, StateActionTransition<S> >::const_iterator item =
168  _model.begin();
169  std::advance(item, rand() % _model.size());
170 
171  const StateActionTransition<S>& st = _model.at(item->first);
172 
173  const S& transState = st.getNextState();
174 
175  A nextAction = argMax(transState, actionSet);
176 
177  backUpStateActionPair(item->first, st.getReward(transState),
178  StateAction<S, A>(transState, nextAction));
179  }
180 }
